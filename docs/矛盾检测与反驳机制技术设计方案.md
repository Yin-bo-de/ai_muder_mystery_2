# AI案件解谜应用 - 矛盾检测与反驳机制技术设计方案

## 1. 概述

### 1.1 设计目标
实现PRD中要求的核心交互功能：
- **矛盾检测系统**：自动识别对话中的时间线、空间关系、证据矛盾
- **智能反驳机制**：AI角色根据威胁程度、反驳证据、人设因素决定是否插话反驳
- **前端体验增强**：@嫌疑人功能、打字机效果、系统提示样式、单独审讯增强

### 1.2 核心设计原则
- **高质量实现**：追求最佳用户体验，智能自然的交互逻辑
- **有限反驳**：每轮对话最多2次反驳，避免无止境对话
- **智能决策**：威胁程度 + 反驳证据 + 人设因素综合判断
- **混合方案**：预定义矛盾点 + LLM轻量级分析

## 3. 后端核心模块设计

### 3.1 矛盾检测引擎（contradiction_detector.py）

**位置：** `backend/app/agents/contradiction_detector.py`

**核心职责：**
1. 分析对话历史，匹配预定义矛盾点
2. 生成用户系统提示
3. 判断是否满足AI角色反驳触发条件

**核心方法：**

```python
class ContradictionDetector:
    """矛盾检测引擎"""

    def __init__(self, case: Case):
        self.case = case
        self.contradiction_points = case.contradiction_points

    async def analyze_dialogue(
        self,
        dialogue_history: List[Message],
        latest_responses: List[Dict[str, Any]]
    ) -> Tuple[List[str], List[Dict[str, Any]]]:
        """
        分析对话，检测矛盾点
        :param dialogue_history: 完整对话历史
        :param latest_responses: 本轮最新的嫌疑人回复
        :return: (系统提示列表, 可触发反驳的矛盾点列表)
        """
        system_hints = []
        refutable_contradictions = []

        for contradiction in self.contradiction_points:
            # 检查是否满足触发条件
            if self._check_trigger_condition(contradiction, latest_responses):
                # 添加用户系统提示
                system_hints.append(contradiction.hint_for_user)

                # 检查是否可以触发反驳
                if self._can_trigger_refusal(contradiction):
                    refutable_contradictions.append({
                        "contradiction": contradiction,
                        "target_suspect": self._get_refusal_target(contradiction),
                        "refuting_suspect": self._get_refuting_suspect(contradiction)
                    })

        return system_hints, refutable_contradictions

    def _check_trigger_condition(
        self,
        contradiction: ContradictionPoint,
        latest_responses: List[Dict[str, Any]]
    ) -> bool:
        """检查矛盾点是否满足触发条件"""
        condition = contradiction.trigger_condition

        # 检查涉及的嫌疑人是否都发言了
        if condition.get("requires_both_speaking", False):
            involved_speakers = {
                resp["suspect_id"] for resp in latest_responses
                if resp["suspect_id"] in contradiction.involved_suspects
            }
            if len(involved_speakers) < len(contradiction.involved_suspects):
                return False

        # 检查关键词
        keywords = condition.get("keywords", [])
        if keywords:
            all_text = " ".join([
                resp["content"] for resp in latest_responses
            ]).lower()
            has_keywords = any(
                keyword.lower() in all_text
                for keyword in keywords
            )
            if not has_keywords:
                return False

        return True

    def _can_trigger_refusal(self, contradiction: ContradictionPoint) -> bool:
        """检查是否可以触发AI角色之间的反驳"""
        # 只有特定类型的矛盾才触发反驳
        if contradiction.type in [
            ContradictionType.TIMELINE,
            ContradictionType.EVIDENCE
        ]:
            return True
        return False

    def _get_refusal_target(self, contradiction: ContradictionPoint) -> str:
        """获取被反驳的目标嫌疑人"""
        # 通常是第一个发言的嫌疑人
        return contradiction.involved_suspects[0]

    def _get_refuting_suspect(self, contradiction: ContradictionPoint) -> str:
        """获取进行反驳的嫌疑人"""
        # 通常是第二个发言的嫌疑人
        if len(contradiction.involved_suspects) > 1:
            return contradiction.involved_suspects[1]
        return contradiction.involved_suspects[0]

    def generate_contradiction_clue(
        self,
        contradiction: ContradictionPoint
    ) -> Optional[Clue]:
        """
        生成矛盾点对应的线索（供用户勘查发现）
        """
        if not contradiction.related_clue_id or not contradiction.clue_location:
            return None

        return Clue(
            clue_id=contradiction.related_clue_id,
            name=f"{contradiction.description}的线索",
            clue_type=ClueType.TESTIMONY,
            status=ClueStatus.HIDDEN,
            description=contradiction.description,
            location=contradiction.clue_location,
            scene=contradiction.clue_location,
            related_suspects=contradiction.involved_suspects,
            importance=0.8
        )
```

### 3.2 反驳决策引擎（refusal_decision_engine.py）

**位置：** `backend/app/agents/refusal_decision_engine.py`

**核心职责：**
1. 评估威胁程度
2. 检查反驳证据
3. 考虑人设因素
4. 限制反驳次数
5. 做出最终决策

**核心方法：**

```python
class ThreatLevel(float, Enum):
    """威胁程度等级"""
    LOW = 0.3
    MEDIUM = 0.6
    HIGH = 0.85
    CRITICAL = 0.95

class RefusalDecisionEngine:
    """反驳决策引擎"""

    def __init__(self, suspects: List[Suspect]):
        self.suspects = {s.suspect_id: s for s in suspects}
        self.max_refusals_per_round = 2

    async def make_refusal_decision(
        self,
        contradiction: Dict[str, Any],
        current_refusal_count: int,
        dialogue_history: List[Message]
    ) -> Tuple[bool, Optional[str]]:
        """
        做出是否反驳的决策
        :param contradiction: 矛盾点信息
        :param current_refusal_count: 当前轮次已反驳次数
        :param dialogue_history: 对话历史
        :return: (是否反驳, 反驳理由提示词)
        """
        # 1. 检查反驳次数限制
        if current_refusal_count >= self.max_refusals_per_round:
            return False, None

        target_suspect_id = contradiction["target_suspect"]
        refuting_suspect_id = contradiction["refuting_suspect"]
        contradiction_point = contradiction["contradiction"]

        # 2. 获取嫌疑人信息
        target_suspect = self.suspects.get(target_suspect_id)
        refuting_suspect = self.suspects.get(refuting_suspect_id)

        if not target_suspect or not refuting_suspect:
            return False, None

        # 3. 评估威胁程度
        threat_score = self._assess_threat_level(
            contradiction_point,
            refuting_suspect
        )

        # 4. 检查反驳证据
        evidence_score = self._check_counter_evidence(
            contradiction_point,
            refuting_suspect
        )

        # 5. 考虑人设因素
        personality_modifier = refuting_suspect.personality_modifier

        # 6. 计算最终得分
        final_score = threat_score * evidence_score * personality_modifier

        # 7. 与阈值比较
        threshold = refuting_suspect.refusal_threshold

        logger.info(
            f"反驳决策 - 嫌疑人: {refuting_suspect.name}, "
            f"威胁分: {threat_score:.2f}, 证据分: {evidence_score:.2f}, "
            f"人设修正: {personality_modifier:.2f}, 最终分: {final_score:.2f}, 阈值: {threshold:.2f}"
        )

        if final_score >= threshold:
            # 生成反驳提示词
            refusal_prompt = self._generate_refusal_prompt(
                contradiction_point,
                target_suspect,
                refuting_suspect
            )
            return True, refusal_prompt

        return False, None

    def _assess_threat_level(
        self,
        contradiction: ContradictionPoint,
        refuting_suspect: Suspect
    ) -> float:
        """
        评估威胁程度（0-1）
        """
        # 根据矛盾类型评估
        if contradiction.type == ContradictionType.EVIDENCE:
            # 被出示关键证据 - 高威胁
            return ThreatLevel.CRITICAL

        elif contradiction.type == ContradictionType.TIMELINE:
            # 时间线被质疑 - 中高威胁
            return ThreatLevel.HIGH

        elif contradiction.type == ContradictionType.SPATIAL:
            # 空间关系矛盾 - 中等威胁
            return ThreatLevel.MEDIUM

        # 默认中等威胁
        return ThreatLevel.MEDIUM

    def _check_counter_evidence(
        self,
        contradiction: ContradictionPoint,
        refuting_suspect: Suspect
    ) -> float:
        """
        检查反驳证据充分度（0-1）
        """
        # 检查counter_evidence中是否有相关证据
        contradiction_key = contradiction.description[:20]  # 取前20字作为key

        for key, evidence in refuting_suspect.counter_evidence.items():
            if key in contradiction.description or contradiction_key in key:
                # 有确凿证据
                return 1.0

        # 检查是否有相关线索
        # (简化实现，实际可以更复杂)
        return 0.6  # 默认有一定理由可以反驳

    def _generate_refusal_prompt(
        self,
        contradiction: ContradictionPoint,
        target_suspect: Suspect,
        refuting_suspect: Suspect
    ) -> str:
        """生成反驳提示词"""
        return f"""
你是{refuting_suspect.name}，你听到{target_suspect.name}的话，发现了矛盾。

请根据你的人设，进行插话反驳：
- 保持你的性格特点：{', '.join(refuting_suspect.personality_traits)}
- 反驳要自然，符合对话场景
- 可以指出对方的矛盾，但不要太咄咄逼人
- 反驳内容控制在1-2句话

矛盾点：{contradiction.description}
你可以使用的反驳理由：{refuting_suspect.counter_evidence}
"""
```

```python
class DialogueManager:
    # ... 现有代码 ...

    async def process_user_message(
        self,
        message: str,
        target_suspects: Optional[List[str]] = None,
        is_accusation: bool = False,
        evidence_shown: Optional[str] = None,
    ) -> Tuple[List[Dict[str, Any]], List[str]]:
        """
        处理用户消息，增强版：包含矛盾检测和反驳机制
        """
        logger.info(f"处理用户消息: {message}, 目标嫌疑人: {target_suspects}")

        # 1. 重置反驳计数（每轮用户消息开始时）
        self._reset_refusal_count()

        # 2. 分析用户意图（现有代码）
        intent, intent_targets = self._analyze_user_intent(message)

        # 3. 确定回复优先级（现有代码）
        priority_list = self._determine_reply_priority(
            message, target_suspects, is_accusation
        )

        # 4. 生成嫌疑人回复（现有代码）
        responses = []
        for suspect_id, priority in priority_list:
            agent = self.suspect_agents.get(suspect_id)
            if not agent:
                continue

            try:
                response = await agent.respond(
                    message=message,
                    is_being_accused=is_accusation and suspect_id in (target_suspects or []),
                    evidence_shown=evidence_shown
                )
                response["suspect_id"] = suspect_id
                response["priority"] = priority
                responses.append(response)
            except Exception as e:
                logger.error(f"嫌疑人 {suspect_id} 生成回复失败: {e}")

        # 5. 新增：矛盾检测
        system_hints = []
        refutable_contradictions = []
        if self.contradiction_detector:
            system_hints, refutable_contradictions = await self.contradiction_detector.analyze_dialogue(
                self._get_dialogue_history(),
                responses
            )

        # 6. 新增：处理反驳
        refusal_responses = []
        if self.refusal_decision_engine and refutable_contradictions:
            for contradiction_info in refutable_contradictions:
                # 检查是否应该反驳
                should_refuse, refusal_prompt = await self.refusal_decision_engine.make_refusal_decision(
                    contradiction_info,
                    self.refusal_count,
                    self._get_dialogue_history()
                )

                if should_refuse and refusal_prompt:
                    # 生成反驳回复
                    refuting_suspect_id = contradiction_info["refuting_suspect"]
                    agent = self.suspect_agents.get(refuting_suspect_id)

                    if agent:
                        refusal_response = await agent.respond_with_prompt(
                            refusal_prompt,
                            message_type="refusal"
                        )
                        refusal_response["suspect_id"] = refuting_suspect_id
                        refusal_response["is_refusal"] = True
                        refusal_responses.append(refusal_response)

                        # 增加反驳计数
                        self.refusal_count += 1

                        # 达到上限停止
                        if self.refusal_count >= self.max_refusals_per_round:
                            break

        # 7. 合并回复
        all_responses = responses + refusal_responses

        # 8. 检测矛盾（现有代码，保留作为补充）
        contradiction_hints = self._detect_contradictions(all_responses)
        system_hints.extend(contradiction_hints)

        logger.info(f"生成 {len(all_responses)} 条嫌疑人回复，系统提示 {len(system_hints)} 条")
        return all_responses, system_hints

    def _reset_refusal_count(self):
        """重置反驳计数"""
        now = datetime.utcnow()
        # 每2分钟或新用户消息重置
        if (now - self.last_refusal_reset) > timedelta(minutes=2):
            self.refusal_count = 0
            self.last_refusal_reset = now

    def _get_dialogue_history(self) -> List[Message]:
        """获取对话历史（从session_service获取）"""
        # 简化实现，实际从session获取
        return []

    async def _generate_contradiction_points(self, case: Case) -> List[ContradictionPoint]:
        """生成案件的矛盾点"""
        contradiction_points = []

        # 生成2-3个矛盾点
        num_contradictions = random.randint(2, 3)

        for i in range(num_contradictions):
            # 选择2个嫌疑人
            suspects_pair = random.sample(case.suspects, 2)

            # 随机选择矛盾类型
            contradiction_type = random.choice([
                ContradictionType.TIMELINE,
                ContradictionType.SPATIAL,
                ContradictionType.EVIDENCE
            ])

            # 生成矛盾描述
            description, hint = self._create_contradiction_description(
                contradiction_type, suspects_pair
            )

            contradiction = ContradictionPoint(
                contradiction_id=f"contradiction_{i+1:03d}",
                type=contradiction_type,
                description=description,
                involved_suspects=[s.suspect_id for s in suspects_pair],
                trigger_condition={
                    "requires_both_speaking": True,
                    "keywords": self._extract_keywords(description)
                },
                hint_for_user=hint,
                related_clue_id=f"c{len(case.clues)+i+1:03d}",
                clue_location=f"{suspects_pair[0].name}的{random.choice(['卧室', '办公室', '房间'])}"
            )

            contradiction_points.append(contradiction)

            # 同时为嫌疑人添加counter_evidence
            for suspect in suspects_pair:
                suspect.counter_evidence = self._generate_counter_evidence(
                    suspect, contradiction
                )
                suspect.personality_modifier = self._calculate_personality_modifier(suspect)

        return contradiction_points

    async def _add_contradiction_clues(self, case: Case):
        """为矛盾点生成对应的线索"""
        detector = ContradictionDetector(case)

        for contradiction in case.contradiction_points:
            clue = detector.generate_contradiction_clue(contradiction)
            if clue:
                case.clues.append(clue)

    def _create_contradiction_description(
        self,
        contradiction_type: ContradictionType,
        suspects_pair: List[Suspect]
    ) -> Tuple[str, str]:
        """创建矛盾描述和用户提示"""
        s1_name = suspects_pair[0].name
        s2_name = suspects_pair[1].name

        if contradiction_type == ContradictionType.SPATIAL:
            description = f"{s2_name}的卧室窗户能看到{s1_name}所在的区域"
            hint = f"【{s2_name}的卧室窗户能看到{s1_name}所在的区域，但{s2_name}说没看到任何人？】"
        elif contradiction_type == ContradictionType.TIMELINE:
            time = random.choice(["19:00", "20:00", "21:00"])
            description = f"{s1_name}和{s2_name}对{time}的时间线描述不一致"
            hint = f"【{s1_name}和{s2_name}对{time}的时间线描述有出入？】"
        else:
            description = f"{s1_name}的说法与某件证据相矛盾"
            hint = f"【{s1_name}的说法似乎与某些证据不太一致？】"

        return description, hint

    def _generate_counter_evidence(
        self,
        suspect: Suspect,
        contradiction: ContradictionPoint
    ) -> Dict[str, str]:
        """为嫌疑人生成反驳证据库"""
        return {
            contradiction.description[:30]: f"我可以证明自己当时确实在{suspect.alibi}",
            "被质疑": "我有不在场证明"
        }

    def _calculate_personality_modifier(self, suspect: Suspect) -> float:
        """根据性格计算修正系数"""
        # 简单的关键词匹配
        personality_str = str(suspect.personality_traits).lower()

        if any(word in personality_str for word in ["暴躁", "冲动", "易怒"]):
            return 1.5
        elif any(word in personality_str for word in ["冷静", "谨慎", "稳重"]):
            return 0.7
        elif any(word in personality_str for word in ["紧张", "心虚"]):
            return 1.2
        else:
            return 1.0

    def _extract_keywords(self, text: str) -> List[str]:
        """从文本中提取关键词"""
        # 简单实现
        words = text.replace("，", " ").replace("。", " ").split()
        return [w for w in words if len(w) >= 2]
```

### 4.2 打字机效果组件

**组件位置：** `frontend/src/components/TypewriterText.tsx`

**技术方案：**
- React useState + setTimeout实现
- 支持可配置速度
- 支持跳过动画

**实现代码：**

```tsx
import { useState, useEffect, useRef, useCallback } from 'react'

interface TypewriterTextProps {
  text: string
  speed?: number  // 毫秒/字符
  onComplete?: () => void
  mood?: string
}

const TypewriterText = ({
  text,
  speed = 30,
  onComplete,
  mood
}: TypewriterTextProps) => {
  const [displayText, setDisplayText] = useState('')
  const [isComplete, setIsComplete] = useState(false)
  const [isSkipped, setIsSkipped] = useState(false)
  const indexRef = useRef(0)
  const timeoutRef = useRef<NodeJS.Timeout | null>(null)

  // 根据mood调整速度
  const getSpeed = useCallback(() => {
    let baseSpeed = speed
    if (mood === 'nervous' || mood === 'scared') {
      baseSpeed *= 0.7  // 紧张时语速快
    } else if (mood === 'calm') {
      baseSpeed *= 1.2  // 冷静时语速慢
    }
    return baseSpeed
  }, [speed, mood])

  const typeNextChar = useCallback(() => {
    if (indexRef.current >= text.length) {
      setIsComplete(true)
      onComplete?.()
      return
    }

    setDisplayText(text.slice(0, indexRef.current + 1))
    indexRef.current += 1

    timeoutRef.current = setTimeout(typeNextChar, getSpeed())
  }, [text, getSpeed, onComplete])

  const skipAnimation = useCallback(() => {
    if (timeoutRef.current) {
      clearTimeout(timeoutRef.current)
    }
    setDisplayText(text)
    setIsComplete(true)
    setIsSkipped(true)
    onComplete?.()
  }, [text, onComplete])

  useEffect(() => {
    // 重置
    indexRef.current = 0
    setDisplayText('')
    setIsComplete(false)
    setIsSkipped(false)

    // 开始打字
    timeoutRef.current = setTimeout(typeNextChar, 100)

    return () => {
      if (timeoutRef.current) {
        clearTimeout(timeoutRef.current)
      }
    }
  }, [text, typeNextChar])

  return (
    <span
      onClick={!isComplete ? skipAnimation : undefined}
      style={{
        cursor: !isComplete ? 'pointer' : 'default'
      }}
    >
      {displayText}
      {!isComplete && !isSkipped && (
        <span
          style={{
            display: 'inline-block',
            width: '0.6em',
            animation: 'blink 1s step-end infinite'
          }}
        >
          |
        </span>
      )}
      <style>{`
        @keyframes blink {
          50% { opacity: 0; }
        }
      `}</style>
    </span>
  )
}

export default TypewriterText
```

| 决策项 | 选择方案 | 理由 |
|--------|----------|------|
| 矛盾检测方式 | 预定义矛盾点 + LLM轻量级分析 | 保证确定性，同时保留灵活性 |

| 前端打字机 | React useState + setTimeout | 简单可靠，用户体验好 |
| @嫌疑人功能 | Ant Design Mentions组件 | 开箱即用，交互体验好 |
| 矛盾点存储 | Case模型 + 对应线索 | 混合方案，系统可用，用户可发现 |
| 反驳次数限制 | 每轮最多2次 | 保证用户体验，避免无止境对话 |
